Call_once Flag

Description

The call once flag ensures that a callable is only called once. The "call_once()" function take a once_flag and callable function as parameters.

Example

#include <iostream>
#include <mutex>
#include <time.h>
#include <chrono>

using namespace std;

once_flag flag;
string x;

void get_time(){
    time_t     now = time(0);
    struct tm  tstruct;
    char       buf[80];
    tstruct = *localtime(&now);
    strftime(buf, sizeof(buf), "%Y-%m-%d.%X", &tstruct);
    
    x = buf;
}

int main(){
    
    call_once(flag,get_time);
    cout << x << endl;
    call_once(flag,get_time);
    cout << x << endl;
    call_once(flag,get_time);
    cout << x << endl;
    
    return 0;
}
