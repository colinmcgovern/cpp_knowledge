Virtual

Description

Virtual functions are functions which are defined at runtime. They differ from normal functions, because normal function are defined at compile time. Because virtual functions have late binding, they are overridabe and are able to be used for polymorphism.

Example

#include <iostream>

using namespace std;

class base{
    public:
        virtual void print_name(){
            cout << "base" << endl;
        };
        
        void print_name_non_virtual(){
           cout << "base" << endl;
        };
};

class derived : public base{
    
    public:
        void print_name(){
            cout << "derived" << endl;
        };  
        
        void print_name_non_virtual(){
           cout << "derived" << endl;
        };
};

int main(){

    derived d;
    base* bptr = &d;
    
    bptr->print_name(); //derived
    bptr->print_name_non_virtual(); //base
    bptr->base::print_name_non_virtual(); //base

    return 0;
}
This program outputs: 

derived
base
base
